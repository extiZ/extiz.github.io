Задание 1 4 способа внедрения JS в HTML
1

Встроенный обработчик событий

onclick="..."

JS-код записывается прямо в атрибут HTML-тега (onclick, onmouseover и др.). Срабатывает при наступлении события. Подходит для простых одноразовых действий. 

<!-- Способ 1: inline-обработчик -->
<button onclick="var box = document.getElementById('res');
    box.innerHTML = 'Время: ' + new Date().toLocaleTimeString();
    box.style.display = 'block';">
    Нажми меня
</button>
2

Внутренний скрипт в <head>

<script> в head

Тег <script> размещается внутри <head>. Используется для объявления функций, доступных до рендеринга страницы. Прямой доступ к DOM возможен только после DOMContentLoaded. 

<!-- В <head> -->
<script>
  function factorial(n) {
    if (n <= 1) return 1;
    return n * factorial(n - 1);
  }
  function runHeadScript() {
    var n = parseInt(document.getElementById('head-input').value);
    document.getElementById('result-head')
      .innerHTML = n + '! = ' + factorial(n);
  }
</script>
3

Внутренний скрипт в <body>

<script> в body

Тег <script> размещается в конце <body>. Это рекомендуемая практика: DOM уже построен, страница отображается до выполнения JS, скрипт не блокирует рендеринг. 

<!-- В конце <body> -->
<script>
  function runBodyScript() {
    var nums = [3, 7, 1, 15, 4, 9, 2];
    var sum  = nums.reduce(function(a,b){ return a+b; }, 0);
    var max  = Math.max.apply(null, nums);
    document.getElementById('result-body')
      .innerHTML = 'Сумма: ' + sum + ' | Макс: ' + max;
  }
</script>
4

Внешний JS-файл

src="external.js"

JS-код выносится в отдельный файл external.js и подключается через <script src="...">. Атрибут defer откладывает выполнение до полной загрузки DOM. Лучший подход для реальных проектов: разделение ответственности, кеширование, переиспользование. 

<!-- Подключение в <head> -->
<script src="external.js" defer></script>

<!-- Содержимое external.js -->
function runExternalScript() {
  var now = new Date();
  document.getElementById('result-external')
    .innerHTML = 'Время: ' + now.toLocaleTimeString();
}
Задание 2 Операторы JS: if, switch, for, while, do..while, break, continue, return
А

Оператор if / else if / else + return

Ветвление

if проверяет условие и выполняет нужную ветку. return немедленно прерывает выполнение функции и возвращает значение. 

function checkGrade(score) {
  if (isNaN(score)) return 'Ошибка';   // ← return

  if (score >= 90)      return '5 — Отлично';
  else if (score >= 75) return '4 — Хорошо';
  else if (score >= 60) return '3 — Удовл.';
  else                  return '2 — Неудовл.';
}
Б

Оператор switch + break

Множественный выбор

switch сравнивает значение с несколькими вариантами (case). break завершает выполнение текущего блока — без него выполнение «провалится» в следующий case. 

switch (month) {
  case 12: case 1: case 2:
    season = 'Зима'; break;   // ← break
  case 3: case 4: case 5:
    season = 'Весна'; break;
  case 6: case 7: case 8:
    season = 'Лето'; break;
  case 9: case 10: case 11:
    season = 'Осень'; break;
  default:
    season = 'Неизвестно';
}
В

Оператор for + continue + break

Цикл со счётчиком

for — цикл с заданным числом итераций. continue пропускает текущую итерацию и переходит к следующей. break полностью выходит из цикла при выполнении условия. 

for (var i = 1; i <= 30; i++) {
  if (i % 2 === 0) continue; // ← пропускаем чётные
  collected.push(i);
  sum += i;
  if (sum >= 50) break;       // ← выходим при сумме ≥ 50
}
Г

Оператор while + break

Цикл с предусловием

while проверяет условие перед каждой итерацией. Здесь строится последовательность Коллаца: цикл продолжается, пока число не станет равным 1. break используется как защита от зависания при слишком длинной последовательности. 

var x = n, steps = 0;
while (x !== 1) {             // ← while
  x = (x % 2 === 0) ? x/2 : 3*x+1;
  steps++;
  if (steps > 300) break;  // ← break (защита)
}
Д

Оператор do..while

Цикл с постусловием

do..while выполняет тело цикла как минимум один раз, а условие проверяется после каждой итерации — в отличие от while. Здесь находим первую степень двойки, превышающую заданное число. 

var power = 1, exp = 0;
do {                    // ← тело выполняется хотя бы раз
  exp++;
  power *= 2;
} while (power < target); // ← условие ПОСЛЕ тела
// Результат: 2^exp ≥ target
Задание 3 Диалоговые окна: alert(), confirm(), prompt()
1

Метод window.alert()

Сообщение

alert(сообщение) показывает модальное окно с текстом и единственной кнопкой «OK». Используется для уведомлений пользователя. Выполнение скрипта приостанавливается до закрытия окна. Возвращает undefined. 

// window можно не указывать явно
alert('Привет!\nЭто окно alert().\nДокумент: ' + document.title);
// После закрытия — продолжаем работу
2

Метод window.confirm()

Подтверждение

confirm(вопрос) показывает модальное окно с двумя кнопками: «OK» и «Отмена». Возвращает true при нажатии OK и false при нажатии Отмена или закрытии окна. 

var answer = confirm('Вы хотите видеть текущее время?');
if (answer) {
  // пользователь нажал OK → true
  console.log('Время: ' + new Date().toLocaleTimeString());
} else {
  // пользователь нажал Отмена → false
  console.log('Отказано.');
}
3

Метод window.prompt()

Ввод данных

prompt(вопрос, значение_по_умолчанию) показывает окно с полем ввода. Возвращает введённую строку при нажатии OK или null при нажатии Отмена. 

var name = prompt('Введите ваше имя:', 'Иван');

if (name === null) {
  // ← нажата кнопка «Отмена»
  console.log('Ввод отменён');
} else if (name.trim() === '') {
  // ← OK нажата, но строка пустая
  console.log('Имя не введено');
} else {
  console.log('Привет, ' + name + '!');
}
